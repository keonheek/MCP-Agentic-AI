### Basic Setup

```python
# pip install discord.py anthropic

import discord
import anthropic

client = discord.Client(intents=discord.Intents.default())
claude = anthropic.Anthropic(api_key="YOUR_KEY")

@client.event
async def on_message(message):
    if message.author == client.user:
        return
    
    response = claude.messages.create(
        model="claude-opus-4-5",
        max_tokens=1024,
        messages=[{"role": "user", "content": message.content}]
    )
    
    await message.channel.send(response.content[0].text)

client.run("YOUR_DISCORD_TOKEN")
```

### Tool Execution Loop

```python
def handle_tool_call(tool_name, tool_input):
    if tool_name == "read_file":
        with open(tool_input["path"], "r") as f:
            return f.read()
    elif tool_name == "web_search":
        # plug in Serper, Tavily, etc.
        return search(tool_input["query"])

# If Claude wants to use a tool, execute it and feed result back
while response.stop_reason == "tool_use":
    tool_use = next(b for b in response.content if b.type == "tool_use")
    result = handle_tool_call(tool_use.name, tool_use.input)
    
    # Send result back to Claude
    response = claude.messages.create(
        model="claude-opus-4-5",
        tools=tools,
        messages=[
            {"role": "user", "content": message.content},
            {"role": "assistant", "content": response.content},
            {"role": "user", "content": [{"type": "tool_result", "tool_use_id": tool_use.id, "content": result}]}
        ]
    )
```
